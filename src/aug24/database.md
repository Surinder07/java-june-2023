SELECT EMPLOYEE_NAME, EMPLOYEE_SALARY FROM EMPLOYEE

-- Sort in order ,  by default in acending order
SELECT EMPLOYEE_NAME, EMPLOYEE_SALARY
FROM EMPLOYEE
ORDER BY EMPLOYEE_SALARY ;

-- Descending order
SELECT EMPLOYEE_NAME, EMPLOYEE_SALARY
FROM EMPLOYEE
ORDER BY EMPLOYEE_SALARY DESC;
